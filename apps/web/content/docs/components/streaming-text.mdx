title: Streaming Text
description: Display text with an animated cursor while streaming from AI.

A simple component that shows a blinking cursor at the end of text while content is being streamed from an AI model. Supports smooth character-by-character rendering for a typewriter effect.

```tsx title="streaming-text.tsx"
"use client";

import * as React from "react";
import { cn } from "@/lib/utils";

interface StreamingTextProps {
  children: string;
  streaming?: boolean;
  cursor?: string;
  smooth?: boolean;
  speed?: number;
  className?: string;
  cursorClassName?: string;
}

function useSmoothText(
  text: string,
  enabled: boolean,
  charsPerTick = 2,
  tickMs = 16,
) {
  const [displayedLength, setDisplayedLength] = React.useState(0);
  const targetLengthRef = React.useRef(0);
  const rafRef = React.useRef<number | null>(null);
  const lastTickRef = React.useRef(0);

  React.useEffect(() => {
    targetLengthRef.current = text.length;

    if (!enabled) {
      setDisplayedLength(text.length);
      return;
    }

    const animate = (timestamp: number) => {
      if (timestamp - lastTickRef.current >= tickMs) {
        lastTickRef.current = timestamp;
        setDisplayedLength((prev) => {
          const next = Math.min(prev + charsPerTick, targetLengthRef.current);
          return next;
        });
      }

      if (displayedLength < targetLengthRef.current) {
        rafRef.current = requestAnimationFrame(animate);
      }
    };

    rafRef.current = requestAnimationFrame(animate);

    return () => {
      if (rafRef.current) {
        cancelAnimationFrame(rafRef.current);
      }
    };
  }, [text, enabled, charsPerTick, tickMs, displayedLength]);

  React.useEffect(() => {
    if (text.length < displayedLength) {
      setDisplayedLength(0);
    }
  }, [text, displayedLength]);

  return enabled ? text.slice(0, displayedLength) : text;
}

export function StreamingText({
  children,
  streaming = false,
  cursor = "▌",
  smooth = false,
  speed = 2,
  className,
  cursorClassName,
}: StreamingTextProps) {
  const displayedText = useSmoothText(children, smooth, speed);
  const isAnimating = smooth && displayedText.length < children.length;

  return (
    <span className={cn("whitespace-pre-wrap", className)}>
      {displayedText}
      {(streaming || isAnimating) && (
        <span
          className={cn("ml-0.5 inline-block animate-pulse", cursorClassName)}
          aria-hidden="true"
        >
          {cursor}
        </span>
      )}
    </span>
  );
}
```

## Usage

### With useObject

```tsx
"use client";

import { experimental_useObject as useObject } from "@ai-sdk/react";
import { StreamingText } from "@/components/ui/streaming-text";

export function StreamingResponse() {
  const { object, isLoading, submit } = useObject({
    api: "/api/stream/text",
    schema: yourSchema,
  });

  return (
    <StreamingText streaming={isLoading} smooth>
      {object?.text ?? ""}
    </StreamingText>
  );
}
```

### With useCompletion

```tsx
"use client";

import { useCompletion } from "@ai-sdk/react";
import { StreamingText } from "@/components/ui/streaming-text";

export function ChatResponse() {
  const { completion, isLoading } = useCompletion({
    api: "/api/completion",
  });

  return (
    <StreamingText streaming={isLoading}>
      {completion}
    </StreamingText>
  );
}
```

### With useChat

```tsx
"use client";

import { useChat } from "@ai-sdk/react";
import { StreamingText } from "@/components/ui/streaming-text";

export function ChatMessage({ message, isStreaming }) {
  return (
    <div className="rounded-lg border p-4">
      <StreamingText streaming={isStreaming}>
        {message.content}
      </StreamingText>
    </div>
  );
}
```

## Examples

### Smooth Typewriter Effect

Enable smooth character-by-character rendering for a typewriter effect:

```tsx
<StreamingText streaming={isLoading} smooth>
  {text}
</StreamingText>
```

### Custom Speed

Control how many characters appear per animation frame:

```tsx
<StreamingText streaming={isLoading} smooth speed={4}>
  {text}
</StreamingText>
```

### Custom Cursor

```tsx
<StreamingText streaming={isLoading} cursor="●">
  {text}
</StreamingText>
```

### Custom Styling

```tsx
<StreamingText 
  streaming={isLoading}
  className="font-mono text-sm"
  cursorClassName="text-blue-500"
>
  {text}
</StreamingText>
```

### Block Cursor

```tsx
<StreamingText 
  streaming={isLoading}
  cursor="█"
  cursorClassName="animate-blink"
>
  {text}
</StreamingText>
```

Add custom blink animation in your CSS:

```css
@keyframes blink {
  0%, 50% { opacity: 1; }
  51%, 100% { opacity: 0; }
}

.animate-blink {
  animation: blink 1s step-end infinite;
}
```
